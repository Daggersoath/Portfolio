<script type="text/javascript">
    var xmlHttp = null;
    
    xmlHttp = new XMLHttpRequest();
    
    xmlHttp.open( "GET", "http://alextheprog.me/Scripts/RTS_Slideshow.html", true );
    xmlHttp.send( null );
    
    if(xmlHttp.status === 200 || xmlHttp.status === 304)
    {
        document.getElementById("leftContent").innerHTML = xmlHttp.responseText;
        document.location.hash = location.substr(0, location.lastIndexOf("."));
    }
    else
    {
        document.write("Failed to load images!");
    }
</script>
